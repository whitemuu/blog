- C :: control
- S :: shift
- M :: option
- s :: command
- X :: combo modifiers(S-S-M-s)
- L :: Leaders (S-S-M-s-<f1~13>)
- E :: Escape
- x|y :: where x = tap, y = hold
- n :: layer n modifier, where n ∈ [1 2 3 4]
